#include <iostream>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <string.h>
#include <unistd.h>

using namespace std;

const char *PATH_FIFO = "/home/artem/Programming/OC/lab6/Task2/task_fifo";

int main(int argc, char **argv)
{
    if (argc > 1)
    {
        if (!strcmp(argv[1], "1"))
        {

            int fd;

            string message;
            char buffer[1024];
            while (true)
            {
                fd = open(PATH_FIFO, O_WRONLY);
                cout << "Enter message: ";
                getline(cin, message);
                write(fd, message.c_str(), message.length());
                close(fd);
                fd = open(PATH_FIFO, O_RDONLY);
                int bytes = read(fd, buffer, sizeof(buffer));
                cout << "Received message: " << string(buffer, bytes) << endl;
                close(fd);
            }
            cout << "[+] Closing connection" << endl;
        }

        if (!strcmp(argv[1], "2"))
        {
            int fd1;

            string message;
            char buffer[1024];
            while (true)
            {
                fd1 = open(PATH_FIFO, O_RDONLY);
                int bytes = read(fd1, buffer, sizeof(buffer));
                cout << "Received message: " << string(buffer, bytes) << endl;
                close(fd1);
                
                fd1 = open(PATH_FIFO, O_WRONLY);
                cout << "Enter message: ";
                getline(cin, message);
                write(fd1, message.c_str(), message.length());
                close(fd1);
            }
            cout << "[+] Closing connection" << endl;
        }
    }
    return 0;
}
